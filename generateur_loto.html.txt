    <button onclick="genererTirages()">Générer les Tirages</button>

    <h2>Résultats :</h2>
    <ul id="resultats"></ul>

    <script>
        function genererTirages() {
            const tiragesCount = document.getElementById('tirages-count').value;
            const resultatsElement = document.getElementById('resultats');
            resultatsElement.innerHTML = '';  // Clear previous results

            for (let i = 0; i < tiragesCount; i++) {
                // Choisir un premier numéro aléatoire entre 1 et 49
                const premierNumero = Math.floor(Math.random() * 49) + 1;

                // Calculer les autres numéros
                const deuxiemeNumero = (premierNumero * 5) - 5;
                const troisiemeNumero = premierNumero + deuxiemeNumero;
                const quatriemeNumero = (premierNumero * 5) + 5;
                const cinquiemeNumero = quatriemeNumero + 5;

                // Vérifier que tous les numéros sont entre 1 et 49
                if (
                    deuxiemeNumero < 1 || deuxiemeNumero > 49 ||
                    troisiemeNumero < 1 || troisiemeNumero > 49 ||
                    quatriemeNumero < 1 || quatriemeNumero > 49 ||
                    cinquiemeNumero < 1 || cinquiemeNumero > 49
                ) {
                    i--;  // Relancer si un numéro est invalide
                    continue;
                }

                // Créer un tirage unique
                const tirage = [premierNumero, deuxiemeNumero, troisiemeNumero, quatriemeNumero, cinquiemeNumero].sort((a, b) => a - b);

                // Numéro chance = 1
                const numeroChance = 1;

                // Ajouter le tirage aux résultats
                const tirageString = `${tirage.join(' - ')} + [Chance : ${numeroChance}]`;
                const li = document.createElement('li');
                li.textContent = tirageString;
                resultatsElement.appendChild(li);
            }
        }
    </script>
